HTML:

    <div id='zoom' class='imBox-container'></div>

    <div id='zoomContainer'>
        <li class='thumbnail' data-imboxsrc='...'>&bullet;</li>
        <li class='thumbnail' data-imboxsrc='...'>&bullet;</li>
    </div>

JS:

    var params = {
        container:document.getElementById('zoom'),
        images:document.querySelectorAll('#zoomContainer>li'),
        imagesIsThumbnails:true,
        zoomOnClick:true
    };
    var thumbnails = new ImBox(params);

3. Slider with thumbnails

HTML:

    <div id='thumbnails' class='imBox-container'></div>

    <div id='thumbnailsContainer'>
        <img class='thumbnail' src='...' data-imboxsrc='...'>
        <img class='thumbnail' src='...' data-imboxsrc='...'>
    </div>

JS:

    var params = {
        container:document.querySelector('#thumbnails'),
        images:document.querySelectorAll('#thumbnailsContainer>img'),
        imagesIsThumbnails:true,
        zoomOnClick:false
    };
    var thumbnails = new ImBox(params);

SCSS:

    #thumbnailsContainer{
        .thumbnail{
            border:2px solid #fff;
            cursor:pointer;
            &.active{
                border:2px solid tomato;
            }
        }
    }
